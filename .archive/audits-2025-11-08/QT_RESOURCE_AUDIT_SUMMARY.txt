Pattern 1: Try/Finally for QTimer (Exemplary)

try:
    timer = QTimer()
    timer.setSingleShot(True)
    test_timers.append(timer)
    timer.start(50)
finally:
    for timer in test_timers:
        timer.stop()
        try:
            timer.timeout.disconnect()
        except RuntimeError:
            pass
        timer.deleteLater()
